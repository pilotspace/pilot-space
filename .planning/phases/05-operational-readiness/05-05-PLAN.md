    ```python
    import os
    import subprocess
    from pathlib import Path
    from urllib.parse import urlparse

    def _extract_password(database_url: str) -> str:
        return urlparse(database_url).password or ""

    def pg_dump(database_url: str, output_path: Path) -> None:
        """Run pg_dump in custom format. Password via PGPASSWORD env (not CLI arg)."""
        env = {**os.environ, "PGPASSWORD": _extract_password(database_url)}
        subprocess.run(
            ["pg_dump", "--format=custom", "--no-password", "--file", str(output_path), database_url],
            env=env,
            check=True,
            capture_output=True,
            text=True,
        )

    def pg_restore(database_url: str, dump_path: Path) -> None:
        """Run pg_restore with --clean --if-exists for fresh database compatibility."""
        env = {**os.environ, "PGPASSWORD": _extract_password(database_url)}
        subprocess.run(
            ["pg_restore", "--clean", "--if-exists", "--no-password", "--dbname", database_url, str(dump_path)],
            env=env,
            check=True,
            capture_output=True,
            text=True,
        )
    ```

    ```python
    from __future__ import annotations

    import sys
    from pathlib import Path

    import typer
    from rich.console import Console
    from rich.progress import Progress, SpinnerColumn, TextColumn

    from pilot_cli.backup.archive import create_archive, extract_archive, read_manifest
    from pilot_cli.backup.encryption import decrypt_file, encrypt_file
    from pilot_cli.backup.pg_backup import pg_dump, pg_restore
    from pilot_cli.backup.storage_backup import download_storage_objects
    from pilot_cli.config import load_config

    backup_app = typer.Typer(name="backup", help="Backup and restore all workspace data.")
    console = Console()


    @backup_app.command("create")
    def create_backup(
        output: Path = typer.Option(Path("./backups"), "--output", "-o", help="Output directory for archive"),
        workspace: str | None = typer.Option(None, "--workspace", help="Backup specific workspace slug (default: all)"),
        encrypt: bool = typer.Option(False, "--encrypt", help="Encrypt archive with AES-256-GCM"),
        passphrase: str | None = typer.Option(None, "--passphrase", envvar="BACKUP_PASSPHRASE", help="Encryption passphrase"),
    ) -> None:
        """Create a backup archive of PostgreSQL and Supabase Storage."""
        config = load_config()
        output.mkdir(parents=True, exist_ok=True)

        if encrypt and not passphrase:
            passphrase = typer.prompt("Encryption passphrase", hide_input=True, confirmation_prompt=True)

        with Progress(SpinnerColumn(), TextColumn("[progress.description]{task.description}"), console=console) as progress:
            # pg_dump
            task = progress.add_task("Dumping PostgreSQL...", total=None)
            import tempfile, datetime
            with tempfile.TemporaryDirectory() as tmp:
                tmp_path = Path(tmp)
                dump_path = tmp_path / "postgres.dump"
                try:
                    pg_dump(config.database_url, dump_path)
                except Exception as e:
                    console.print(f"[red]pg_dump failed: {e}[/red]")
                    raise typer.Exit(1)
                progress.update(task, description="PostgreSQL dump complete")

                # Storage download
                progress.update(task, description="Downloading Supabase Storage objects...")
                objects = download_storage_objects(
                    supabase_url=config.supabase_url,
                    service_key=config.supabase_service_key,
                    workspace=workspace,
                )
                progress.update(task, description=f"Downloaded {len(objects)} storage objects")

                # Create archive
                progress.update(task, description="Creating archive...")
                timestamp = datetime.datetime.now(datetime.UTC).strftime("%Y%m%dT%H%M%SZ")
                archive_name = f"pilot-space-backup-{timestamp}.tar.gz"
                archive_path = output / archive_name
                create_archive(dump_path, objects, archive_path)

                # Encrypt if requested
                if encrypt:
                    progress.update(task, description="Encrypting archive...")
                    encrypted_path = archive_path.with_suffix(".tar.gz.enc")
                    encrypt_file(archive_path, encrypted_path, passphrase)
                    archive_path.unlink()
                    archive_path = encrypted_path

        console.print(f"[green]Backup complete:[/green] {archive_path}")


    @backup_app.command("restore")
    def restore_backup(
        archive: Path = typer.Argument(..., help="Path to backup archive (.tar.gz or .tar.gz.enc)"),
        dry_run: bool = typer.Option(False, "--dry-run", help="Validate archive without restoring"),
        passphrase: str | None = typer.Option(None, "--passphrase", envvar="BACKUP_PASSPHRASE"),
    ) -> None:
        """Restore workspace data from a backup archive."""
        if not archive.exists():
            console.print(f"[red]Archive not found: {archive}[/red]")
            raise typer.Exit(1)

        # Decrypt if needed
        actual_archive = archive
        if archive.suffix == ".enc":
            if not passphrase:
                passphrase = typer.prompt("Decryption passphrase", hide_input=True)
            import tempfile
            tmp_dir = Path(tempfile.mkdtemp())
            actual_archive = tmp_dir / "decrypted.tar.gz"
            decrypt_file(archive, actual_archive, passphrase)

        manifest = read_manifest(actual_archive)
        console.print("[bold]Archive manifest:[/bold]")
        console.print(f"  Created: {manifest.get('created_at')}")
        console.print(f"  PostgreSQL dump: {manifest.get('pg_dump_file')}")
        console.print(f"  Storage objects: {manifest.get('storage_objects_count', 0)}")

        if dry_run:
            console.print("[green]Dry run complete — archive is valid[/green]")
            return

        config = load_config()
        console.print(f"\n[yellow]Target database:[/yellow] {config.database_url[:40]}...")
        confirmed = typer.confirm("This will overwrite the target database. Continue?")
        if not confirmed:
            raise typer.Exit(0)

        import tempfile
        with tempfile.TemporaryDirectory() as tmp:
            tmp_path = Path(tmp)
            extract_archive(actual_archive, tmp_path)
            dump_path = tmp_path / manifest["pg_dump_file"]
            try:
                pg_restore(config.database_url, dump_path)
            except Exception as e:
                console.print(f"[red]pg_restore failed: {e}[/red]")
                raise typer.Exit(1)

        console.print("[green]Restore complete[/green]")
    ```
