Update detection logic (from RESEARCH.md, string inequality only):
```typescript
// In MCPCatalogStore or as a utility function:
function hasUpdate(entry: McpCatalogEntry, server: MCPServer): boolean {
  if (!server.installed_catalog_version) return false;
  return server.installed_catalog_version !== entry.catalog_version;
}

function isInstalled(entry: McpCatalogEntry, installedServers: MCPServer[]): boolean {
  return installedServers.some(s => s.catalog_entry_id === entry.id);
}
```

MCPCatalogStore pattern (MobX, mirrors MCPServersStore):
```typescript
export class MCPCatalogStore {
  entries: McpCatalogEntry[] = [];
  isLoading = false;
  error: string | null = null;

  constructor() { makeAutoObservable(this); }

  async loadCatalog(): Promise<void> {
    runInAction(() => { this.isLoading = true; this.error = null; });
    try {
      const data = await mcpCatalogApi.list();
      runInAction(() => { this.entries = data.items; });
    } catch (err) {
      runInAction(() => { this.error = err instanceof Error ? err.message : 'Failed to load catalog'; });
    } finally {
      runInAction(() => { this.isLoading = false; });
    }
  }
}
```
