```python
from uuid import UUID
from sqlalchemy import select, and_
from sqlalchemy.ext.asyncio import AsyncSession
from pilot_space.domain.artifact_annotation import ArtifactAnnotation

class ArtifactAnnotationRepository:
    def __init__(self, session: AsyncSession) -> None:
        self._session = session

    async def list_by_slide(self, artifact_id: UUID, slide_index: int) -> list[ArtifactAnnotation]:
        stmt = (
            select(ArtifactAnnotation)
            .where(and_(
                ArtifactAnnotation.artifact_id == artifact_id,
                ArtifactAnnotation.slide_index == slide_index,
            ))
            .order_by(ArtifactAnnotation.created_at.asc())
        )
        result = await self._session.execute(stmt)
        return list(result.scalars().all())

    async def create(self, annotation: ArtifactAnnotation) -> ArtifactAnnotation:
        self._session.add(annotation)
        await self._session.flush()
        return annotation

    async def get_by_id(self, annotation_id: UUID) -> ArtifactAnnotation | None:
        return await self._session.get(ArtifactAnnotation, annotation_id)

    async def delete(self, annotation: ArtifactAnnotation) -> None:
        await self._session.delete(annotation)
        await self._session.flush()
```

**4. Service** — `backend/src/pilot_space/application/services/artifact/artifact_annotation_service.py`:

```python
from dataclasses import dataclass
from uuid import UUID
from sqlalchemy.ext.asyncio import AsyncSession
from pilot_space.domain.artifact_annotation import ArtifactAnnotation
from pilot_space.infrastructure.repositories.artifact_annotation_repository import ArtifactAnnotationRepository

class AnnotationNotFoundError(Exception):
    error_code = "annotation_not_found"
    http_status = 404
    def __init__(self, message: str = "Annotation not found") -> None:
        super().__init__(message)
        self.message = message

class AnnotationForbiddenError(Exception):
    error_code = "annotation_forbidden"
    http_status = 403
    def __init__(self, message: str = "Not allowed to modify this annotation") -> None:
        super().__init__(message)
        self.message = message

@dataclass
class CreateAnnotationPayload:
    artifact_id: UUID
    workspace_id: UUID
    user_id: UUID
    slide_index: int
    content: str

@dataclass
class UpdateAnnotationPayload:
    annotation_id: UUID
    user_id: UUID
    content: str

class ArtifactAnnotationService:
    def __init__(self, session: AsyncSession) -> None:
        self._repo = ArtifactAnnotationRepository(session)

    async def list_by_slide(self, artifact_id: UUID, slide_index: int) -> list[ArtifactAnnotation]:
        return await self._repo.list_by_slide(artifact_id, slide_index)

    async def create(self, payload: CreateAnnotationPayload) -> ArtifactAnnotation:
        annotation = ArtifactAnnotation(
            artifact_id=payload.artifact_id,
            workspace_id=payload.workspace_id,
            user_id=payload.user_id,
            slide_index=payload.slide_index,
            content=payload.content,
        )
        return await self._repo.create(annotation)

    async def update(self, payload: UpdateAnnotationPayload) -> ArtifactAnnotation:
        annotation = await self._repo.get_by_id(payload.annotation_id)
        if not annotation:
            raise AnnotationNotFoundError()
        if annotation.user_id != payload.user_id:
            raise AnnotationForbiddenError()
        annotation.content = payload.content
        await self._repo._session.flush()
        return annotation

    async def delete(self, annotation_id: UUID, user_id: UUID) -> None:
        annotation = await self._repo.get_by_id(annotation_id)
        if not annotation:
            raise AnnotationNotFoundError()
        if annotation.user_id != user_id:
            raise AnnotationForbiddenError()
        await self._repo.delete(annotation)
```

**1. Create `usePptxAnnotations.ts`:**

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiClient } from '@/lib/api-client';

interface Annotation {
  id: string;
  artifact_id: string;
  user_id: string;
  slide_index: number;
  content: string;
  created_at: string;
  updated_at: string;
  isPending?: boolean;
}

export const annotationKeys = {
  all: ['annotations'] as const,
  list: (artifactId: string, slideIndex: number) =>
    ['annotations', artifactId, slideIndex] as const,
};

export function usePptxAnnotations(artifactId: string, slideIndex: number) {
  const queryClient = useQueryClient();
  const queryKey = annotationKeys.list(artifactId, slideIndex);

  const { data: annotations = [], isLoading, isError } = useQuery({
    queryKey,
    queryFn: async () => {
      const res = await apiClient.get<Annotation[]>(
        `/api/v1/artifacts/${artifactId}/annotations?slide_index=${slideIndex}`
      );
      return res.data;
    },
    staleTime: 1000 * 60, // 1 minute
  });

  const createMutation = useMutation({
    mutationFn: async (data: { content: string }) => {
      const res = await apiClient.post<Annotation>(
        `/api/v1/artifacts/${artifactId}/annotations`,
        { slide_index: slideIndex, content: data.content }
      );
      return res.data;
    },
    onMutate: async (newData) => {
      await queryClient.cancelQueries({ queryKey });
      const previous = queryClient.getQueryData<Annotation[]>(queryKey);
      queryClient.setQueryData<Annotation[]>(queryKey, (old) => [
        ...(old ?? []),
        {
          id: `temp-${Date.now()}`,
          artifact_id: artifactId,
          user_id: '',
          slide_index: slideIndex,
          content: newData.content,
          created_at: new Date().toISOString(),
          updated_at: new Date().toISOString(),
          isPending: true,
        },
      ]);
      return { previous };
    },
    onError: (_err, _new, context) => {
      queryClient.setQueryData(queryKey, context?.previous);
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey });
    },
  });

  const updateMutation = useMutation({
    mutationFn: async (data: { annotationId: string; content: string }) => {
      const res = await apiClient.put<Annotation>(
        `/api/v1/artifacts/${artifactId}/annotations/${data.annotationId}`,
        { content: data.content }
      );
      return res.data;
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey });
    },
  });

  const deleteMutation = useMutation({
    mutationFn: async (annotationId: string) => {
      await apiClient.delete(
        `/api/v1/artifacts/${artifactId}/annotations/${annotationId}`
      );
    },
    onMutate: async (annotationId) => {
      await queryClient.cancelQueries({ queryKey });
      const previous = queryClient.getQueryData<Annotation[]>(queryKey);
      queryClient.setQueryData<Annotation[]>(queryKey, (old) =>
        (old ?? []).map((a) =>
          a.id === annotationId ? { ...a, isPending: true } : a
        )
      );
      return { previous };
    },
    onError: (_err, _id, context) => {
      queryClient.setQueryData(queryKey, context?.previous);
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey });
    },
  });

  return {
    annotations,
    isLoading,
    isError,
    createAnnotation: createMutation.mutate,
    updateAnnotation: updateMutation.mutate,
    deleteAnnotation: deleteMutation.mutate,
    isCreating: createMutation.isPending,
  };
}
```

**Add `formatRelativeTime` helper** (inline or imported):
```typescript
function formatRelativeTime(iso: string): string {
  const diff = Date.now() - new Date(iso).getTime();
  const mins = Math.floor(diff / 60000);
  if (mins < 1) return 'just now';
  if (mins < 60) return `${mins}m ago`;
  const hrs = Math.floor(mins / 60);
  if (hrs < 24) return `${hrs}h ago`;
  const days = Math.floor(hrs / 24);
  return `${days}d ago`;
}
```
