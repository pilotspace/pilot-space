The test targets the `additional_tools` argument passed to `configure_sdk_for_space`. Since `_build_stream_config` is an async method on `PilotSpaceAgent` with many dependencies, test the logic in isolation by extracting it: mock `_load_remote_mcp_servers` and `configure_sdk_for_space`, call `_build_stream_config` (or test the pattern-generation logic directly as a pure unit), then assert on the captured `additional_tools` kwarg.

Because `_build_stream_config` has heavy DI dependencies, the cleanest unit approach is to test the wildcard-generation expression directly as a pure function: `remote_tool_patterns = [f"mcp__{key}__*" for key in remote_servers]` and then `ALL_TOOL_NAMES + remote_tool_patterns`. This is the exact logic being added — test it deterministically without mocking the full agent.

Test file structure:
```python
"""Unit tests for MCPI-01: remote MCP tool wildcard pattern generation in _build_stream_config."""
from __future__ import annotations

from pilot_space.ai.agents.pilotspace_agent_helpers import ALL_TOOL_NAMES


def _generate_remote_tool_patterns(remote_servers: dict) -> list[str]:
    """Mirrors the production expression added to _build_stream_config."""
    return [f"mcp__{key}__*" for key in remote_servers]


def test_no_remote_servers_produces_no_patterns() -> None:
    patterns = _generate_remote_tool_patterns({})
    assert patterns == []
    combined = ALL_TOOL_NAMES + patterns
    assert combined == ALL_TOOL_NAMES


def test_single_remote_server_produces_one_wildcard() -> None:
    remote_servers = {"remote_abc-123-def": object()}
    patterns = _generate_remote_tool_patterns(remote_servers)
    assert patterns == ["mcp__remote_abc-123-def__*"]


def test_multiple_remote_servers_produce_one_wildcard_each() -> None:
    remote_servers = {"remote_aaa": object(), "remote_bbb": object()}
    patterns = _generate_remote_tool_patterns(remote_servers)
    assert "mcp__remote_aaa__*" in patterns
    assert "mcp__remote_bbb__*" in patterns
    assert len(patterns) == 2


def test_all_tool_names_preserved_after_merge() -> None:
    remote_servers = {"remote_xyz": object()}
    patterns = _generate_remote_tool_patterns(remote_servers)
    combined = ALL_TOOL_NAMES + patterns
    for name in ALL_TOOL_NAMES:
        assert name in combined
```
