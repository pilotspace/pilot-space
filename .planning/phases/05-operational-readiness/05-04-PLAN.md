    # Backend service
    backend:
      replicaCount: 3
      image:
        repository: pilot-space/backend
        tag: latest
        pullPolicy: Always
      resources:
        requests:
          cpu: 100m
          memory: 256Mi
        limits:
          cpu: 500m
          memory: 512Mi
      env: {}  # Extra env vars as key/value map

    # Frontend service
    frontend:
      replicaCount: 2
      image:
        repository: pilot-space/frontend
        tag: latest
        pullPolicy: Always
      resources:
        requests:
          cpu: 50m
          memory: 128Mi
        limits:
          cpu: 200m
          memory: 256Mi

    # Ingress configuration
    ingress:
      enabled: true
      className: nginx
      host: pilot-space.example.com
      tls:
        enabled: true
        secretName: pilot-space-tls

    # External dependencies (enterprises use managed services)
    externalDatabase:
      host: ""
      port: 5432
      name: pilot_space
      existingSecret: ""      # Secret must have key: database-url
      existingSecretKey: database-url

      host: ""
      port: 6379
      existingSecret: ""      # Secret must have key: redis-url

      existingSecretAnonKey: anon-key
      existingSecretServiceKey: service-key
      existingSecretJwtSecret: jwt-secret

    # HPA configuration
    autoscaling:
      enabled: true
      backend:
        minReplicas: 2
        maxReplicas: 10
        targetCPUUtilizationPercentage: 70
      frontend:
        minReplicas: 2
        maxReplicas: 5
        targetCPUUtilizationPercentage: 70

      minAvailable: 1

    Create infra/helm/pilot-space/templates/_helpers.tpl with standard Helm helpers:
    ```
    {{/*
    Expand the name of the chart.
    */}}
    {{- define "pilot-space.name" -}}
    {{- default .Chart.Name .Values.nameOverride | trunc 63 | trimSuffix "-" }}
    {{- end }}

    {{/*
    Create a default fully qualified app name.
    */}}
    {{- define "pilot-space.fullname" -}}
    {{- if .Values.fullnameOverride }}
    {{- .Values.fullnameOverride | trunc 63 | trimSuffix "-" }}
    {{- else }}
    {{- $name := default .Chart.Name .Values.nameOverride }}
    {{- printf "%s-%s" .Release.Name $name | trunc 63 | trimSuffix "-" }}
    {{- end }}
    {{- end }}

    {{/*
    Common labels
    */}}
    {{- define "pilot-space.labels" -}}
    helm.sh/chart: {{ include "pilot-space.name" . }}-{{ .Chart.Version }}
    app.kubernetes.io/name: {{ include "pilot-space.name" . }}
    app.kubernetes.io/instance: {{ .Release.Name }}
    app.kubernetes.io/version: {{ .Chart.AppVersion | quote }}
    app.kubernetes.io/managed-by: {{ .Release.Service }}
    {{- end }}

    {{/*
    Selector labels
    */}}
    {{- define "pilot-space.selectorLabels" -}}
    app.kubernetes.io/name: {{ include "pilot-space.name" . }}
    app.kubernetes.io/instance: {{ .Release.Name }}
    {{- end }}
    ```

    templates/deployment-backend.yaml — critical requirements:
    - image: "{{ .Values.backend.image.repository }}:{{ .Values.backend.image.tag }}"
    - readinessProbe using httpGet path: /health/ready port: 8000 (from OPS-03, plan 05-01)
    - livenessProbe using httpGet path: /health/live port: 8000
    - initialDelaySeconds: 15 for liveness, 10 for readiness (allow startup time)
    - env vars sourced from existingSecret references (NOT values.yaml literals):
      ```yaml
      env:
        - name: DATABASE_URL
          valueFrom:
            secretKeyRef:
              name: {{ .Values.externalDatabase.existingSecret }}
              key: {{ .Values.externalDatabase.existingSecretKey }}
        - name: REDIS_URL
          valueFrom:
            secretKeyRef:
              name: {{ .Values.externalRedis.existingSecret }}
              key: {{ .Values.externalRedis.existingSecretKey }}
        - name: SUPABASE_URL
          value: {{ .Values.externalSupabase.url | quote }}
        - name: SUPABASE_ANON_KEY
          valueFrom:
            secretKeyRef:
              name: {{ .Values.externalSupabase.existingSecret }}
              key: {{ .Values.externalSupabase.existingSecretAnonKey }}
      ```
    - resources from {{ .Values.backend.resources }}
    - replicas: {{ .Values.backend.replicaCount }}
    - strategy: RollingUpdate with maxSurge: 1, maxUnavailable: 0

    templates/deployment-frontend.yaml — similar pattern, image from .Values.frontend.image, no DB secrets needed.

    templates/ingress.yaml — conditional on {{ .Values.ingress.enabled }}:
    ```yaml
    {{- if .Values.ingress.enabled }}
    apiVersion: networking.k8s.io/v1
    kind: Ingress
    metadata:
      name: {{ include "pilot-space.fullname" . }}
      annotations:
        nginx.ingress.kubernetes.io/ssl-redirect: "true"
    spec:
      ingressClassName: {{ .Values.ingress.className }}
      rules:
        - host: {{ .Values.ingress.host }}
          http:
            paths:
              - path: /api
                pathType: Prefix
                backend:
                  service:
                    name: {{ include "pilot-space.fullname" . }}-backend
                    port:
                      number: 8000
              - path: /
                pathType: Prefix
                backend:
                  service:
                    name: {{ include "pilot-space.fullname" . }}-frontend
                    port:
                      number: 3000
      {{- if .Values.ingress.tls.enabled }}
      tls:
        - hosts:
            - {{ .Values.ingress.host }}
          secretName: {{ .Values.ingress.tls.secretName }}
      {{- end }}
    {{- end }}
    ```
